#defining a function to fill age value
def input_age(col):
    Age = col[0]
    Pclass = col[1]
    if pd.isnull(Age):
        if Pclass == 1:
            return 37 #this is avg value of age of 1st class passengers
        elif Pclass==2:
            return 29
        else:
            return 25
        
    else:
        return Age
In [18]:
